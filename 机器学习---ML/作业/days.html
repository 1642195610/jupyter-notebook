def age_cut(col):
    if col >= 20 and col < 40:
        return "年轻"
    elif col < 60:
        return "中年"
    elif col < 80:
        return "中老年"
    else:
        return "老年"

def drbm(data,col: str):
    prefix = col + "_"
    df = pd.get_dummies(data[col],prefix = prefix)
    da = pd.concat((data,df),axis = 1,ignore_index = False)
    return da

for i in col:
    patients = drbm(patients,i)
